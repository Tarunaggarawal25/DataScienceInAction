            guess_df = titanice_dataset[(titanice_dataset['Sex'] == i) & \
                                  (titanice_dataset['Pclass'] == j+1)]['Age'].dropna()

            # age_mean = guess_df.mean()

            age_guess = guess_df.median()

            # Convert random age float to nearest .5 age
            guess_ages[i,j] = int( age_guess/0.5 + 0.5 ) * 0.5

        for j in range(0, 3):
            titanice_dataset.loc[ (titanice_dataset.Age.isnull()) & (titanice_dataset.Sex == i) & (titanice_dataset.Pclass == j+1),\
